function Car(make, model, year) { this.make = make; this.model = model; this.year = year; this.start = function() { console.log(`Starting the ${this.year} ${this.make} ${this.model}.`); }; } const car1 = new Car("Toyota", "Camry", 2022); const car2 = new Car("Honda", "Civic", 2023); car1.start(); // Output: Starting the 2022 Toyota Camry. car2.start(); // Output: Starting the 2023 Honda Civic.
